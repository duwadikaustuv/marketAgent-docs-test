**3. Complementary Components**

**Diagram.jsx**

```javascript
import { useEffect, useRef } from "react";
import mermaid from "mermaid";

mermaid.initialize({
  theme: "dark",
  securityLevel: "loose",
  flowchart: { curve: "basis" },
});

export const Diagram = ({ children, id }) => {
  const containerRef = useRef();
  const diagramId = id || `mermaid-${Math.random().toString(36).slice(2)}`;

  useEffect(() => {
    const render = async () => {
      try {
        const { svg } = await mermaid.render(diagramId, children);
        containerRef.current.innerHTML = svg;
      } catch (err) {
        containerRef.current.innerHTML = `<div class="text-red-400">Diagram Error: ${err.message}</div>`;
      }
    };
    render();
  }, [children, diagramId]);

  return (
    <div ref={containerRef} className="my-6 bg-code-base p-4 rounded-lg" />
  );
};
```
